## 3. 数据类型定义

```typescript
// src/types/skill.ts

interface Skill {
  id: string;
  name: string;
  description: string;
  icon: string;              // 图标 URL 或 base64
  category: SkillCategory;
  version: string;
  author: string;
  homepage?: string;
  installed: boolean;
  installedAt?: number;
  config?: SkillConfig;
  features: string[];        // 功能特性列表
  tags?: string[];
}

type SkillCategory =
  | 'data_analysis'
  | 'writing'
  | 'code'
  | 'office'
  | 'communication'
  | 'other';

interface SkillConfig {
  /** 配置项 schema（JSON Schema 格式） */
  schema: Record<string, any>;
  /** 当前配置值 */
  values: Record<string, any>;
}

/** 技能安装请求（发送给 OpenClaw） */
interface InstallSkillRequest {
  skillId: string;
  version: string;
  config?: Record<string, any>;
}
```

## 4. 状态管理

```typescript
// src/stores/useSkillStore.ts

interface SkillState {
  skills: Skill[];
  installedSkills: Skill[];
  isLoading: boolean;
  searchKeyword: string;
  selectedCategory: SkillCategory | 'all';

  // Actions
  fetchSkills: () => Promise<void>;
  installSkill: (skillId: string, config?: Record<string, any>) => Promise<void>;
  uninstallSkill: (skillId: string) => Promise<void>;
  updateSkillConfig: (skillId: string, config: Record<string, any>) => Promise<void>;
  setSearchKeyword: (keyword: string) => void;
  setCategory: (category: SkillCategory | 'all') => void;

  // Computed（通过 selector）
  filteredSkills: Skill[];   // 根据 keyword + category 过滤
}
```

## 5. 关键接口

```typescript
// GET /api/skills               获取技能列表
// GET /api/skills/:id           获取技能详情
// POST /api/skills/:id/install  安装技能
// DELETE /api/skills/:id        卸载技能
// PATCH /api/skills/:id/config  更新技能配置
// GET /api/skills/installed     获取已安装技能列表
```

## 6. 与 OpenClaw 的集成方式

技能安装后，通过两种方式触发使用：
1. **在对话中直接 @技能名**：输入框中输入 `@数据分析师 帮我分析这份数据`
2. **技能详情中"在对话中使用"按钮**：跳转到对话页面并预填提示词

```typescript
// 点击"在对话中使用"后的行为
router.push('/chat');
// 在 ChatStore 中设置预填文本
chatStore.setDraftMessage(`@${skill.name} `);
```
